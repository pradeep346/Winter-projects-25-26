K = 1;
tau = 0.5;
G = tf(K, [tau 1]);


Kp = 2;   
Ki = 1;   
Kd = 0.1; 


C = pid(Kp, Ki, Kd);
T_pid = feedback(C * G, 1); 

figure(2);
step(T_pid);
title('Closed-Loop PID Control Response');
xlabel('Time (seconds)');
ylabel('Amplitude (Speed)');
grid on;

if ~exist('results', 'dir')
    mkdir('results');
end
saveas(gcf, 'results/pid_response.png');

stepinfo(T_pid)
disp(['Steady-State Error: ', num2str(abs(1 - dcgain(T_pid)) * 100), ' %']);
